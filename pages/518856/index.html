# (1) auto (opens new window)

Cpp11ж–°ж ҮеҮҶеј•вјҠдәҶautoзұ»еһӢиҜҙжҳҺз¬ҰпјҢвҪӨе®ғе°ұиғҪи®©зј–иҜ‘еҷЁжӣҝжҲ‘д»¬еҺ»еҲҶжһҗиЎЁиҫҫејҸжүҖеұһзҡ„зұ»еһӢгҖӮе’ҢеҺҹжқҘйӮЈдәӣеҸӘеҜ№еә”жҹҗз§Қ зү№е®ҡзҡ„зұ»еһӢиҜҙжҳҺз¬Ұ(дҫӢеҰӮ int)дёҚеҗҢ

auto и®©зј–иҜ‘еҷЁйҖҡиҝҮеҲқе§ӢеҖјжқҘиҝӣвҫҸзұ»еһӢжҺЁжј”гҖӮд»ҺвҪҪиҺ·еҫ—е®ҡд№үеҸҳйҮҸзҡ„зұ»еһӢпјҢжүҖд»ҘиҜҙ auto е®ҡд№үзҡ„еҸҳйҮҸеҝ…йЎ»жңүеҲқе§ӢеҖјгҖӮ

дёҫдёӘдҫӢвјҰпјҡ

//жҷ®йҖҡпјӣзұ»еһӢ
int a = 1, b = 3;
auto c = a + b;// cдёәintеһӢ

//constзұ»еһӢ
const int i = 5;
auto j = i; // еҸҳйҮҸiжҳҜйЎ¶еұӮconst, дјҡиў«еҝҪз•Ҙ, жүҖд»Ҙjзҡ„зұ»еһӢжҳҜint
auto k = &i; // еҸҳйҮҸiжҳҜвјҖдёӘеёёйҮҸ, еҜ№еёёйҮҸеҸ–ең°еқҖжҳҜвјҖз§Қеә•еұӮconst, жүҖд»Ҙkзҡ„зұ»еһӢжҳҜconst int*
const auto l = i; //еҰӮжһңеёҢжңӣжҺЁж–ӯеҮәзҡ„зұ»еһӢжҳҜйЎ¶еұӮconstзҡ„, йӮЈд№Ҳе°ұйңҖиҰҒеңЁautoеүҚвҫҜеҠ дёҠcosnt

еҢәеҲҶеҫҲз®ҖеҚ•пјҢзңӢconstдҝ®йҘ°зҡ„жҳҜе“ӘдёӘгҖӮеүҚиҖ…дҝ®йҘ°зҡ„жҳҜp1пјҢеҚіжҢҮй’Ҳжң¬иә«пјҢжүҖд»ҘдёәйЎ¶еұӮеұһжҖ§пјӣеҗҺиҖ…дҝ®йҘ°зҡ„*p2пјҢеҚіжҢҮй’Ҳзҡ„жҢҮеҗ‘ең°еқҖзҡ„еҖјпјҢжүҖд»Ҙдёәеә•еұӮеұһжҖ§гҖӮp1еҸҲз§°дёәеёёйҮҸжҢҮй’ҲпјҢp2еҸҲз§°дёәжҢҮеҗ‘еёёйҮҸзҡ„жҢҮй’ҲгҖӮ

  1. йЎ¶еұӮconstеңЁиөӢеҖјз»ҷе…¶д»–еҸҳйҮҸж—¶пјҢеҸҜд»ҘеҝҪз•ҘйЎ¶еұӮеұһжҖ§пјӣ
  2. еә•еұӮconstеңЁиөӢеҖјз»ҷе…¶д»–еҸҳйҮҸж—¶пјҢдёҚиғҪеҝҪз•Ҙеә•еұӮеұһжҖ§пјӣ
  3. int*зұ»еһӢеҸҜд»ҘиҪ¬жҚўдёәйЎ¶еұӮе’Ңеә•еұӮconstпјҢжүҖд»Ҙе®ғеҸҜд»Ҙз»ҷйЎ¶еұӮе’Ңеә•еұӮзҡ„constиөӢеҖјпјӣ
  4. еә•еұӮconstж— жі•иҪ¬жҚўдёәйЎ¶еұӮconstгҖӮ

# autoзҡ„3дёӘжҺЁеҜји§„еҲҷпјҡ

**и§„еҲҷ1пјҡ**еЈ°жҳҺдёәautoпјҲдёҚжҳҜauto&пјүзҡ„еҸҳйҮҸпјҢеҝҪи§ҶжҺүеҲқе§ӢеҢ–иЎЁиҫҫејҸзҡ„йЎ¶еұӮconstгҖӮеҚіеҜ№жңүconstзҡ„жҷ®йҖҡзұ»еһӢ(int гҖҒdoubleзӯү)еҝҪи§ҶconstпјҢеҜ№еёёйҮҸжҢҮй’ҲпјҲйЎ¶еұӮconstпјүеҸҳдёәжҷ®йҖҡжҢҮй’ҲпјҢеҜ№жҢҮеҗ‘еёёйҮҸпјҲеә•еұӮconstпјүзҡ„еёёйҮҸжҢҮй’ҲпјҲйЎ¶еұӮcosntпјүеҸҳдёәжҢҮеҗ‘еёёйҮҸзҡ„жҢҮй’ҲпјҲеә•еұӮconstпјүгҖӮ

**и§„еҲҷ2пјҡ**еЈ°жҳҺдёәauto&зҡ„еҸҳйҮҸпјҢдҝқжҢҒеҲқе§ӢеҢ–иЎЁиҫҫејҸзҡ„йЎ¶еұӮconstжҲ–volatile еұһжҖ§гҖӮ

**и§„еҲҷ3пјҡ****иӢҘеёҢжңӣautoжҺЁеҜјзҡ„жҳҜйЎ¶еұӮconstпјҢеҠ дёҠconstпјҢеҚіconst autoгҖӮ

# (2) decltype

жңүзҡ„ж—¶еҖҷжҲ‘д»¬иҝҳдјҡйҒҮеҲ°иҝҷз§Қжғ…еҶөпјҢжҲ‘д»¬еёҢжңӣд»ҺиЎЁиҫҫејҸдёӯжҺЁж–ӯеҮәиҰҒе®ҡд№үеҸҳйҮҸзҡ„зұ»еһӢпјҢдҪҶеҚҙдёҚжғівҪӨиЎЁиҫҫејҸзҡ„еҖјеҺ»еҲқе§ӢеҢ–еҸҳйҮҸгҖӮиҝҳжңүеҸҜиғҪжҳҜеҮҪж•°зҡ„иҝ”еӣһзұ»еһӢдёәжҹҗиЎЁиҫҫејҸзҡ„еҖјзұ»еһӢгҖӮеңЁиҝҷдәӣж—¶еҖҷautoжҳҫеҫ—е°ұж— еҠӣдәҶпјҢжүҖд»ҘCpp11вјңеј•вјҠдәҶз¬¬вјҶз§Қзұ»еһӢиҜҙжҳҺз¬ҰdecltypeпјҢе®ғзҡ„дҪңвҪӨжҳҜйҖүжӢ©е№¶иҝ”еӣһж“ҚдҪңж•°зҡ„ж•°жҚ®зұ»еһӢгҖӮеңЁжӯӨиҝҮзЁӢдёӯпјҢзј–иҜ‘еҷЁеҸӘжҳҜеҲҶжһҗиЎЁиҫҫејҸе№¶еҫ—еҲ°е®ғзҡ„зұ»еһӢпјҢеҚҙдёҚиҝӣвҫҸе®һйҷ…зҡ„и®Ўз®—иЎЁиҫҫејҸзҡ„еҖјгҖӮ

int func() {return 0};
//жҷ®йҖҡзұ»еһӢ
decltype(func()) sum = 5; // sumзҡ„зұ»еһӢжҳҜеҮҪж•°func()зҡ„иҝ”еӣһеҖјзҡ„зұ»еһӢint, дҪҶжҳҜиҝҷж—¶дёҚдјҡе®һйҷ…и°ғвҪӨеҮҪж•°func()
int a = 0;
decltype(a) b = 4; // aзҡ„зұ»еһӢжҳҜint, жүҖд»Ҙbзҡ„зұ»еһӢд№ҹжҳҜint

//дёҚи®әжҳҜйЎ¶еұӮconstиҝҳжҳҜеә•еұӮconst, decltypeйғҪдјҡдҝқз•ҷ 
const int c = 3;
decltype(c) d = c; // dзҡ„зұ»еһӢе’ҢcжҳҜвјҖж ·зҡ„, йғҪжҳҜйЎ¶еұӮconst
int e = 4;

е®һзҺ°еҺҹзҗҶпјҡunique_ptrйҮҮз”Ёзҡ„жҳҜз§»еҠЁиҜӯд№үпјҢдёҖдёӘйқһз©әзҡ„unique_ptrжҖ»жҳҜжӢҘжңүе®ғжүҖжҢҮеҗ‘зҡ„иө„жәҗгҖӮиҪ¬з§»дёҖдёӘunique_ptrе°ҶдјҡжҠҠжүҖжңүжқғе…ЁйғЁд»ҺжәҗжҢҮй’ҲиҪ¬з§»з»ҷзӣ®ж ҮжҢҮй’ҲпјҢжәҗжҢҮй’Ҳиў«зҪ®з©әпјӣжүҖжңүunique_ptrдёҚж”ҜжҢҒжҷ®йҖҡзҡ„жӢ·иҙқе’ҢиөӢеҖјж“ҚдҪңпјҢдёҚиғҪз”ЁеңЁSTLж ҮеҮҶе®№еҷЁдёӯпјӣеұҖйғЁеҸҳйҮҸзҡ„иҝ”еӣһеҖјйҷӨеӨ–пјҲеӣ дёәзј–иҜ‘еҷЁзҹҘйҒ“иҰҒиҝ”еӣһзҡ„еҜ№иұЎе°ҶиҰҒиў«й”ҖжҜҒпјүпјӣеҰӮжһңдҪ жӢ·иҙқвјҖдёӘ unique_ptrпјҢйӮЈд№ҲжӢ·иҙқз»“жқҹеҗҺпјҢиҝҷдёӨдёӘunique_ptrйғҪдјҡжҢҮеҗ‘зӣёеҗҢзҡ„иө„жәҗпјҢйҖ жҲҗеңЁз»“жқҹж—¶еҜ№еҗҢвјҖеҶ…еӯҳжҢҮй’ҲеӨҡж¬ЎйҮҠж”ҫиҖҢеҜјиҮҙзЁӢеәҸеҙ©жәғгҖӮ

# (3) weak_ptr

weak_ptrпјҡејұеј•з”ЁгҖӮеј•з”Ёи®Ўж•°жңүвјҖдёӘй—®йўҳе°ұжҳҜдә’зӣёеј•з”ЁеҪўжҲҗзҺҜпјҲзҺҜеҪўеј•з”ЁпјүпјҢиҝҷж ·дёӨдёӘжҢҮй’ҲжҢҮеҗ‘зҡ„еҶ…еӯҳйғҪж— жі•йҮҠж”ҫгҖӮйңҖиҰҒдҪҝз”Ёweak_ptrйҖҡиҝҮе°Ҷе…¶дёӯдёҖдёӘжҲ–еӨҡдёӘstd::shared_ptrеҜ№иұЎиҪ¬жҚўдёәstd::weak_ptrеҜ№иұЎпјҢжқҘжү“з ҙзҺҜеҪўеј•з”ЁгҖӮweak_ptrжҳҜвјҖдёӘејұеј•з”ЁпјҢе®ғжҳҜдёәдәҶй…ҚеҗҲshared_ptrиҖҢеј•е…Ҙзҡ„вјҖз§ҚжҷәиғҪжҢҮй’ҲпјҢе®ғжҢҮеҗ‘вјҖдёӘз”ұshared_ptrз®ЎзҗҶзҡ„еҜ№иұЎиҖҢдёҚеҪұе“ҚжүҖжҢҮеҜ№иұЎзҡ„з”ҹе‘Ҫе‘ЁжңҹпјҢд№ҹе°ұжҳҜиҜҙпјҢе®ғеҸӘеј•з”ЁпјҢдёҚи®Ўж•°гҖӮеҰӮжһңвјҖеқ—еҶ…еӯҳиў«shared_ptrе’Ңweak_ptrеҗҢж—¶еј•з”ЁпјҢеҪ“жүҖжңүshared_ptrжһҗжһ„дәҶд№ӢеҗҺпјҢдёҚз®ЎиҝҳжңүжІЎжңүweak_ptrеј•з”ЁиҜҘеҶ…еӯҳпјҢеҶ…еӯҳд№ҹдјҡиў«йҮҠж”ҫгҖӮжүҖд»Ҙweak_ptrдёҚдҝқиҜҒе®ғжҢҮеҗ‘зҡ„еҶ…еӯҳвјҖе®ҡжҳҜжңүж•Ҳзҡ„пјҢеңЁдҪҝз”Ёд№ӢеүҚдҪҝз”ЁеҮҪж•°lock()жЈҖжҹҘweak_ptrжҳҜеҗҰдёәз©әжҢҮй’ҲгҖӮ

йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢstd::weak_ptrеҜ№иұЎдёҚиғҪзӣҙжҺҘи®ҝй—®жүҖз®ЎзҗҶзҡ„еҜ№иұЎпјҢйңҖиҰҒе…Ҳе°Ҷе…¶иҪ¬жҚўдёәstd::shared_ptrеҜ№иұЎжүҚиғҪи®ҝй—®жүҖз®ЎзҗҶзҡ„еҜ№иұЎгҖӮеҰӮжһңеңЁиҪ¬жҚўиҝҮзЁӢдёӯеҸ‘зҺ°иҜҘеҜ№иұЎе·Із»Ҹиў«й”ҖжҜҒпјҢеҲҷиҪ¬жҚўеҗҺзҡ„std::shared_ptrеҜ№иұЎдјҡеҸҳжҲҗnullжҢҮй’ҲгҖӮ

# (4) auto_ptr

дё»иҰҒжҳҜдёәдәҶи§ЈеҶівҖңжңүејӮеёёжҠӣеҮәж—¶еҸ‘з”ҹеҶ…еӯҳжі„жјҸвҖқзҡ„й—®йўҳ гҖӮеӣ дёәеҸ‘з”ҹејӮеёёиҖҢж— жі•жӯЈеёёйҮҠж”ҫеҶ…еӯҳгҖӮ

auto_ptrжңүжӢ·иҙқиҜӯд№үпјҢжӢ·иҙқеҗҺжәҗеҜ№иұЎеҸҳеҫ—ж— ж•ҲпјҢиҝҷеҸҜиғҪеј•еҸ‘еҫҲдёҘйҮҚзҡ„й—®йўҳпјӣиҖҢunique_ptrеҲҷж— жӢ·иҙқиҜӯд№үпјҢдҪҶжҸҗдҫӣдәҶз§»еҠЁиҜӯд№үпјҢиҝҷж ·зҡ„й”ҷиҜҜдёҚеҶҚеҸҜиғҪеҸ‘з”ҹпјҢеӣ дёәеҫҲжҳҺжҳҫеҝ…йЎ»дҪҝз”Ёstd::move()иҝӣиЎҢиҪ¬з§»гҖӮ

auto_ptrдёҚж”ҜжҢҒжӢ·иҙқе’ҢиөӢеҖјж“ҚдҪңпјҢдёҚиғҪз”ЁеңЁSTLж ҮеҮҶе®№еҷЁдёӯгҖӮSTLе®№еҷЁдёӯзҡ„е…ғзҙ з»ҸеёёиҰҒж”ҜжҢҒжӢ·иҙқгҖҒиөӢеҖјж“ҚдҪңпјҢеңЁиҝҷиҝҮзЁӢдёӯauto_ptrдјҡдј йҖ’жүҖжңүжқғпјҢжүҖд»ҘдёҚиғҪеҶҚSTLдёӯдҪҝз”ЁгҖӮ

жҷәиғҪжҢҮй’Ҳshared_ptrд»Јз Ғе®һзҺ°пјҡ

template<typename T>
class SharedPtr
{
public:
 	SharedPtr(T* ptr = NULL):_ptr(ptr), _pcount(new int(1)){}
    
	SharedPtr(const SharedPtr& s):_ptr(s._ptr), _pcount(s._pcount) {
 		(*_pcount)++;
 	}
	SharedPtr<T>& operator=(const SharedPtr& s)
    {
 		if (this != &s)
 		{
 			if (--(*(this->_pcount)) == 0)
 			{
 				delete this->_ptr;
 				delete this->_pcount;
 			}
 			_ptr = s._ptr;
 			_pcount = s._pcount;
 			*(_pcount)++;
 		}
 		return *this;
  	}
 	T& operator*()
 	{
 		return *(this->_ptr);
 	}
	T* operator->()
 	{
 		return this->_ptr;
 	}
 	~SharedPtr()
 	{
 		--(*(this->_pcount));
 		if (*(this->_pcount) == 0)
 		{
 			delete _ptr;
 			_ptr = NULL;
 			delete _pcount;
 			_pcount = NULL;
 		}
 	}
private:
 	T* _ptr;
 	int* _pcount;//жҢҮеҗ‘еј•вҪӨи®Ўж•°зҡ„жҢҮй’Ҳ
};

  4. unique_ptrвҖңе”ҜвјҖвҖқжӢҘжңүе…¶жүҖжҢҮеҜ№иұЎпјҢеҗҢвјҖж—¶еҲ»еҸӘиғҪжңүвјҖдёӘunique_ptrжҢҮеҗ‘з»ҷе®ҡеҜ№иұЎпјҲйҖҡиҝҮзҰҒжӯўжӢ·иҙқиҜӯд№үгҖҒеҸӘжңүз§»еҠЁиҜӯд№үжқҘе®һзҺ°пјүгҖӮзӣёжҜ”дёҺеҺҹе§ӢжҢҮй’Ҳunique_ptrз”ЁдәҺе…¶RAIIзҡ„зү№жҖ§пјҢдҪҝеҫ—еңЁеҮәзҺ°ејӮеёёзҡ„жғ…еҶөдёӢпјҢеҠЁжҖҒиө„жәҗиғҪеҫ—еҲ°йҮҠж”ҫгҖӮunique_ptrжҢҮй’Ҳжң¬иә«зҡ„з”ҹе‘Ҫе‘Ёжңҹпјҡд»Һunique_ptrжҢҮй’ҲеҲӣе»әж—¶ејҖе§ӢпјҢзӣҙеҲ°зҰ»ејҖдҪңз”ЁеҹҹгҖӮзҰ»ејҖдҪңз”Ёеҹҹж—¶пјҢиӢҘе…¶жҢҮеҗ‘еҜ№иұЎпјҢеҲҷе°Ҷе…¶жүҖжҢҮеҜ№иұЎй”ҖжҜҒ(й»ҳи®ӨдҪҝз”Ёdeleteж“ҚдҪңз¬ҰпјҢз”ЁжҲ·еҸҜжҢҮе®ҡе…¶д»–ж“ҚдҪң)гҖӮunique_ptrжҢҮй’ҲдёҺе…¶жүҖжҢҮеҜ№иұЎзҡ„е…ізі»пјҡеңЁжҷәиғҪжҢҮй’Ҳз”ҹе‘Ҫе‘ЁжңҹеҶ…пјҢеҸҜд»Ҙж”№еҸҳжҷәиғҪжҢҮй’ҲжүҖжҢҮеҜ№иұЎпјҢеҰӮеҲӣе»әжҷәиғҪжҢҮй’Ҳж—¶йҖҡиҝҮжһ„йҖ еҮҪж•°жҢҮе®ҡгҖҒйҖҡиҝҮresetвҪ…жі•йҮҚж–°жҢҮе®ҡгҖҒйҖҡиҝҮreleaseж–№жі•йҮҠж”ҫжүҖжңүжқғгҖҒйҖҡиҝҮз§»еҠЁиҜӯд№үиҪ¬з§»жүҖжңүжқғгҖӮ

  5. жҷәиғҪжҢҮй’Ҳзұ»е°ҶвјҖдёӘи®Ўж•°еҷЁдёҺзұ»жҢҮеҗ‘зҡ„еҜ№иұЎзӣёе…іиҒ”пјҢеј•з”Ёи®Ўж•°и·ҹиёӘиҜҘзұ»жңүеӨҡе°‘дёӘеҜ№иұЎе…ұдә«еҗҢвјҖжҢҮй’ҲгҖӮжҜҸж¬ЎеҲӣе»әзұ»зҡ„ж–°еҜ№иұЎж—¶пјҢеҲқе§ӢеҢ–жҢҮй’Ҳе№¶е°Ҷеј•з”Ёи®Ўж•°зҪ®дёә1пјӣеҪ“еҜ№иұЎдҪңдёәеҸҰвјҖеҜ№иұЎзҡ„еүҜжң¬иҖҢеҲӣе»әж—¶пјҢжӢ·иҙқжһ„йҖ еҮҪж•°жӢ·иҙқжҢҮй’Ҳе№¶еўһеҠ дёҺд№Ӣзӣёеә”зҡ„еј•з”Ёи®Ўж•°пјӣеҜ№вјҖдёӘеҜ№иұЎиҝӣиЎҢиөӢеҖјж—¶пјҢиөӢеҖјж“ҚдҪңз¬ҰеҮҸе°‘е·Ұж“ҚдҪңж•°жүҖжҢҮеҜ№иұЎзҡ„еј•з”Ёи®Ўж•°пјҲеҰӮжһңеј•з”Ёи®Ўж•°дёәеҮҸиҮі0пјҢеҲҷеҲ йҷӨеҜ№иұЎпјүпјҢе№¶еўһеҠ еҸіж“ҚдҪңж•°жүҖжҢҮеҜ№иұЎзҡ„еј•з”Ёи®Ўж•°пјӣи°ғз”Ёжһҗжһ„еҮҪж•°ж—¶пјҢжһ„йҖ еҮҪж•°еҮҸе°‘еј•з”Ёи®Ўж•°пјҲеҰӮжһңеј•з”Ёи®Ўж•°еҮҸиҮі0пјҢеҲҷеҲ йҷӨеҹәзЎҖеҜ№иұЎпјүгҖӮ

  6. weak_ptr жҳҜвјҖз§ҚдёҚжҺ§еҲ¶еҜ№иұЎз”ҹе‘Ҫе‘Ёжңҹзҡ„жҷәиғҪжҢҮй’Ҳ, е®ғжҢҮеҗ‘вјҖдёӘ shared_ptr з®ЎзҗҶзҡ„еҜ№иұЎ. иҝӣиЎҢиҜҘеҜ№иұЎзҡ„еҶ…еӯҳз®ЎзҗҶзҡ„жҳҜйӮЈдёӘејәеј•з”Ёзҡ„ shared_ptr. weak_ptrеҸӘжҳҜжҸҗдҫӣдәҶеҜ№з®ЎзҗҶеҜ№иұЎзҡ„вјҖдёӘи®ҝй—®жүӢж®өгҖӮweak_ptr и®ҫи®Ўзҡ„зӣ®зҡ„жҳҜдёәй…ҚеҗҲ shared_ptr иҖҢеј•е…Ҙзҡ„вјҖз§ҚжҷәиғҪжҢҮй’ҲжқҘеҚҸеҠ© shared_ptr е·ҘдҪң, е®ғеҸӘеҸҜд»Ҙд»ҺвјҖдёӘ shared_ptr жҲ–еҸҰвјҖдёӘ weak_ptr еҜ№иұЎжһ„йҖ , е®ғзҡ„жһ„йҖ е’Ңжһҗжһ„дёҚдјҡеј•иө·еј•з”Ёи®°ж•°зҡ„еўһеҠ жҲ–еҮҸе°‘.

# 7гҖҒиҜҙиҜҙдҪ дәҶи§Јзҡ„auto_ptrдҪңз”Ё

  1. auto_ptrзҡ„еҮәзҺ°пјҢдё»иҰҒжҳҜдёәдәҶи§ЈеҶівҖңжңүејӮеёёжҠӣеҮәж—¶еҸ‘з”ҹеҶ…еӯҳжі„жјҸвҖқзҡ„й—®йўҳпјӣжҠӣеҮәејӮеёёпјҢе°ҶеҜјиҮҙжҢҮй’ҲpжүҖжҢҮеҗ‘зҡ„з©әй—ҙеҫ—дёҚеҲ°йҮҠж”ҫиҖҢеҜјиҮҙеҶ…еӯҳжі„жјҸпјӣ
  2. auto_ptrжһ„йҖ ж—¶еҸ–еҫ—жҹҗдёӘеҜ№иұЎзҡ„жҺ§еҲ¶жқғпјҢеңЁжһҗжһ„ж—¶йҮҠж”ҫиҜҘеҜ№иұЎгҖӮжҲ‘д»¬е®һйҷ…дёҠжҳҜеҲӣе»әвјҖдёӘauto_ptrзұ»еһӢзҡ„еұҖйғЁеҜ№иұЎпјҢиҜҘеұҖйғЁеҜ№иұЎжһҗжһ„ж—¶пјҢдјҡе°ҶиҮӘиә«жүҖжӢҘжңүзҡ„жҢҮй’Ҳз©әй—ҙйҮҠж”ҫпјҢжүҖд»ҘдёҚдјҡжңүеҶ…еӯҳжі„жјҸпјӣ
  3. auto_ptrзҡ„жһ„йҖ еҮҪж•°жҳҜexplicitпјҢйҳ»жӯўдәҶвјҖиҲ¬жҢҮй’ҲйҡҗејҸиҪ¬жҚўдёә auto_ptrзҡ„жһ„йҖ пјҢжүҖд»ҘдёҚиғҪзӣҙжҺҘе°ҶвјҖиҲ¬зұ»еһӢзҡ„жҢҮй’ҲиөӢеҖјз»ҷauto_ptrзұ»еһӢзҡ„еҜ№иұЎпјҢеҝ…йЎ»з”Ёauto_ptrзҡ„жһ„йҖ еҮҪж•°еҲӣе»әеҜ№иұЎпјӣ
  4. з”ұдәҺauto_ptrеҜ№иұЎжһҗжһ„ж—¶дјҡеҲ йҷӨе®ғжүҖжӢҘжңүзҡ„жҢҮй’ҲпјҢжүҖд»ҘдҪҝз”Ёж—¶йҒҝе…ҚеӨҡдёӘauto_ptrеҜ№иұЎз®ЎзҗҶеҗҢвјҖдёӘжҢҮй’Ҳпјӣ
  5. Auto_ptrеҶ…йғЁе®һзҺ°пјҢжһҗжһ„еҮҪж•°дёӯеҲ йҷӨеҜ№иұЎз”Ёзҡ„жҳҜdeleteиҖҢдёҚжҳҜdelete[]пјҢжүҖд»Ҙauto_ptrдёҚиғҪз®ЎзҗҶж•°з»„пјӣ
  6. auto_ptrж”ҜжҢҒжүҖжӢҘжңүзҡ„жҢҮй’Ҳзұ»еһӢд№Ӣй—ҙзҡ„йҡҗејҸзұ»еһӢиҪ¬жҚўгҖӮ
  7. еҸҜд»ҘйҖҡиҝҮ*е’Ң->иҝҗз®—з¬ҰеҜ№auto_ptrжүҖжңүз”Ёзҡ„жҢҮй’ҲиҝӣиЎҢжҸҗйўҶж“ҚдҪңпјӣ
  8. T* get(),иҺ·еҫ—auto_ptrжүҖжӢҘжңүзҡ„жҢҮй’ҲпјӣT* release()пјҢйҮҠж”ҫauto_ptrзҡ„жүҖжңүжқғпјҢе№¶е°ҶжүҖжңүз”Ёзҡ„жҢҮй’Ҳиҝ”еӣһгҖӮ
